import { useEffect, useMemo, useState } from "react";

import { computeAvailableSlots, revisitInfo } from "@/lib/availability";
import { ownerHomeCopy } from "@/lib/owner-home-copy";

import type { Appointment, AppointmentStatus, BootstrapPayload, GroomingRecord, Pet, Service } from "@/types/domain";

type TabKey = "home" | "book" | "customers" | "revisit" | "settings";

  | { type: "appointment"; appointment: Appointment }
  | { type: "new-appointment"; petId?: string }
  | { type: "new-customer" }
  | { type: "add-pet"; guardianId: string }
  | { type: "edit-record"; record: GroomingRecord }
  | { type: "stat"; kind: "today" | "pending" | "completed" | "cancel_change" }
  | null;

  pending: { label: "대기", color: "#b45309", bg: "#fef3c7" },
  confirmed: { label: "확정", color: "#047857", bg: "#dcfce7" },
  in_progress: { label: "미용중", color: "#4338ca", bg: "#e0e7ff" },
  almost_done: { label: "거의 완료", color: "#7c3aed", bg: "#ede9fe" },

  cancelled: { label: "취소", color: "#b91c1c", bg: "#fee2e2" },
  rejected: { label: ownerHomeCopy.pendingReject, color: "#b91c1c", bg: "#fee2e2" },
  noshow: { label: "노쇼", color: "#b91c1c", bg: "#fee2e2" },
};

  { key: "home", label: "홈", icon: "홈" },
  { key: "book", label: "예약", icon: "예약" },
  { key: "customers", label: "고객", icon: "고객" },
  { key: "revisit", label: "재방문", icon: "재방문" },
  { key: "settings", label: "설정", icon: "설정" },
];

async function fetchJson<T>(input: RequestInfo, init?: RequestInit) {
  const response = await fetch(input, {
    ...init,
    headers: { "Content-Type": "application/json", ...(init?.headers || {}) },
  });
  const json = await response.json();
  if (!response.ok) throw new Error(json.message || "요청에 실패했습니다.");
  return json as T;
}

  const [selectedDate, setSelectedDate] = useState("2026-03-16");
  const [selectedPetId, setSelectedPetId] = useState<string | null>(null);

  async function refresh() {
    const next = await fetchJson<BootstrapPayload>(`/api/bootstrap?shopId=${data.shop.id}`);
    setData(next);
  }

  useEffect(() => {
    setData(initialData);
  }, [initialData]);

  const serviceMap = useMemo(() => Object.fromEntries(data.services.map((item) => [item.id, item])), [data.services]);
  const guardianMap = useMemo(() => Object.fromEntries(data.guardians.map((item) => [item.id, item])), [data.guardians]);
  const petMap = useMemo(() => Object.fromEntries(data.pets.map((item) => [item.id, item])), [data.pets]);

  const todayConfirmedAppointments = useMemo(() => data.appointments.filter((item) => item.appointment_date === "2026-03-16" && ["confirmed", "in_progress", "almost_done", "completed", "cancelled"].includes(item.status)), [data.appointments]);
  const pendingAppointments = useMemo(() => data.appointments.filter((item) => item.status === "pending"), [data.appointments]);
  const todayActionAppointments = useMemo(() => todayConfirmedAppointments.filter((item) => ["confirmed", "in_progress"].includes(item.status)).sort((a, b) => a.appointment_time.localeCompare(b.appointment_time)), [todayConfirmedAppointments]);
  const todayHistoryAppointments = useMemo(() => todayConfirmedAppointments.filter((item) => ["almost_done", "completed", "cancelled"].includes(item.status)).sort((a, b) => a.appointment_time.localeCompare(b.appointment_time)), [todayConfirmedAppointments]);
  const completedHistoryAppointments = useMemo(() => todayHistoryAppointments.filter((item) => ["almost_done", "completed", "cancelled"].includes(item.status)), [todayHistoryAppointments]);

    const lastRecord = data.groomingRecords.filter((record) => record.pet_id === pet.id).sort((a, b) => b.groomed_at.localeCompare(a.groomed_at))[0];
    const revisit = revisitInfo(pet, lastRecord?.groomed_at);
    return { pet, guardian: guardianMap[pet.guardian_id], lastRecord, ...revisit };
  }).sort((a, b) => (a.daysUntil ?? 999) - (b.daysUntil ?? 999)), [data.groomingRecords, data.pets, guardianMap]);

    const query = customerSearch.trim();
    if (!query) return data.guardians;
    return data.guardians.filter((guardian) => {
      const pets = data.pets.filter((pet) => pet.guardian_id === guardian.id);
      return guardian.name.includes(query) || guardian.phone.includes(query) || pets.some((pet) => `${pet.name}${pet.breed}`.includes(query));
    });
  }, [customerSearch, data.guardians, data.pets]);

  const selectedPet = selectedPetId ? petMap[selectedPetId] : null;
  const selectedGuardian = selectedPet ? guardianMap[selectedPet.guardian_id] : null;
  const selectedRecords = selectedPet ? data.groomingRecords.filter((item) => item.pet_id === selectedPet.id) : [];
  const selectedAppointments = selectedPet ? data.appointments.filter((item) => item.pet_id === selectedPet.id) : [];

    setError(null);

      setError(mutationError instanceof Error ? mutationError.message : "저장에 실패했습니다.");

      setSaving(false);
    }

    await mutate("/api/notifications", {

      body: JSON.stringify({
        shopId: data.shop.id,
        guardianId: guardian.id,
        petId: pet.id,
        type: "revisit_notice",
        message: `${pet.name} 재방문 안내 알림을 발송했어요.`,
      }),
    });
  }

  const overdueCount = revisitRows.filter((item) => item.status === "overdue").length;
  const urgentCount = revisitRows.filter((item) => item.status === "overdue" || item.status === "soon").length;
  const estimatedRevenue = todayConfirmedAppointments.reduce((sum, item) => sum + (serviceMap[item.service_id]?.price || 0), 0);

          </div>

        {activeTab === "book" && <section className="space-y-4 p-4"><Panel title="날짜별 예약"><input type="date" value={selectedDate} onChange={(event) => setSelectedDate(event.target.value)} className="field" /></Panel><Panel title={`${shortDate(selectedDate)} 예약`}>{selectedDayAppointments.length === 0 ? <EmptyState title="선택한 날짜 예약이 없어요" /> : selectedDayAppointments.map((appointment) => <AppointmentRow key={appointment.id} appointment={appointment} pet={petMap[appointment.pet_id]} guardian={guardianMap[appointment.guardian_id]} service={serviceMap[appointment.service_id]} onClick={() => setModal({ type: "appointment", appointment })} />)}</Panel></section>}

        {activeTab === "customers" && selectedPet && <section className="space-y-4 p-4"><button className="text-sm font-bold text-[var(--muted)]" onClick={() => setSelectedPetId(null)}>← 고객 목록</button><Panel title={selectedPet.name} action={`${selectedGuardian?.name} 보호자`}><div className="grid grid-cols-2 gap-3 text-sm"><InfoItem label="견종" value={selectedPet.breed} /><InfoItem label="나이" value={selectedPet.age ? `${selectedPet.age}살` : "미입력"} /><InfoItem label="몸무게" value={selectedPet.weight ? `${selectedPet.weight}kg` : "미입력"} /><InfoItem label="재방문 주기" value={`${selectedPet.grooming_cycle_weeks}주`} /></div>{selectedPet.notes && <div className="mt-3 rounded-2xl bg-[#fff5dd] px-3 py-3 text-sm text-[#8a5d19]">주의사항: {selectedPet.notes}</div>}<div className="mt-4 flex gap-2">{(["info", "records", "appointments"] as const).map((item) => <button key={item} className={`flex-1 rounded-full px-3 py-2 text-xs font-bold ${detailTab === item ? "bg-[var(--accent)] text-white" : "bg-[#f2ece7] text-[var(--muted)]"}`} onClick={() => setDetailTab(item)}>{item === "info" ? "정보" : item === "records" ? `기록 ${selectedRecords.length}` : `예약 ${selectedAppointments.length}`}</button>)}</div>{detailTab === "info" && <button className="mt-4 w-full rounded-2xl bg-[var(--accent)] px-4 py-3 text-sm font-bold text-white" onClick={() => setModal({ type: "new-appointment", petId: selectedPet.id })}>이 반려견으로 예약 등록</button>}{detailTab === "records" && <div className="mt-4 space-y-3">{selectedRecords.length === 0 ? <EmptyState title="미용 기록이 없어요" /> : selectedRecords.map((record) => <RecordCard key={record.id} record={record} service={serviceMap[record.service_id]} onEdit={() => setModal({ type: "edit-record", record })} />)}</div>}{detailTab === "appointments" && <div className="mt-4 space-y-3">{selectedAppointments.length === 0 ? <EmptyState title="예약 이력이 없어요" /> : selectedAppointments.map((appointment) => <AppointmentRow key={appointment.id} appointment={appointment} pet={selectedPet} guardian={selectedGuardian!} service={serviceMap[appointment.service_id]} onClick={() => setModal({ type: "appointment", appointment })} />)}</div>}</Panel></section>}

        {activeTab === "revisit" && <section className="space-y-4 p-4">{(["overdue", "soon", "ok"] as const).map((status) => { const rows = revisitRows.filter((item) => item.status === status); return <Panel key={status} title={status === "overdue" ? "빨강: 초과" : status === "soon" ? "노랑: 5일 이내" : "초록: 여유 있음"}>{rows.length === 0 ? <EmptyState title="해당 고객이 없어요" /> : rows.map((row) => <div key={row.pet.id} className="rounded-3xl border border-[var(--border)] bg-white px-4 py-3"><div className="flex items-center gap-3"><Avatar seed={row.pet.avatar_seed} /><div className="flex-1"><p className="text-sm font-bold">{row.pet.name}</p><p className="text-xs text-[var(--muted)]">{row.guardian?.name} · 마지막 {row.lastRecord ? shortDate(row.lastRecord.groomed_at.slice(0, 10)) : "기록 없음"}</p></div><UrgencyPill status={row.status} days={row.daysUntil} /></div><button className="mt-3 w-full rounded-2xl bg-[var(--accent)] px-3 py-2 text-sm font-bold text-white" onClick={() => sendRevisitNotice(row.pet)}>재방문 알림 로그 남기기</button></div>)}</Panel>; })}</section>}

        {activeTab === "settings" && <SettingsPanel data={data} onSave={async (payload) => mutate("/api/settings", { method: "PATCH", body: JSON.stringify(payload) })} onSaveService={async (payload) => mutate("/api/services", { method: "POST", body: JSON.stringify(payload) })} />}
      </main>

      <nav className="fixed bottom-0 z-20 flex w-full max-w-[430px] border-t border-[var(--border)] bg-white/95 px-2 py-2 backdrop-blur">{tabItems.map((item) => <button key={item.key} className={`flex flex-1 flex-col items-center gap-1 rounded-2xl px-2 py-2 text-[11px] font-bold ${activeTab === item.key ? "bg-[var(--accent-soft)] text-[var(--accent)]" : "text-[var(--muted)]"}`} onClick={() => { setActiveTab(item.key); if (item.key !== "customers") setSelectedPetId(null); }}><span>{item.icon}</span><span>{item.label}</span></button>)}</nav>

      {modal && <div>{modal.type === "appointment" ? <Overlay><AppointmentDetail appointment={modal.appointment} pet={petMap[modal.appointment.pet_id]} guardian={guardianMap[modal.appointment.guardian_id]} service={serviceMap[modal.appointment.service_id]} saving={saving} onClose={() => setModal(null)} onUpdate={(status) => mutate("/api/appointments", { method: "PATCH", body: JSON.stringify({ appointmentId: modal.appointment.id, status }) })} /></Overlay> : null}{modal.type === "new-appointment" ? <Overlay><NewAppointmentForm data={data} petId={modal.petId} saving={saving} onClose={() => setModal(null)} onSave={(payload) => mutate("/api/appointments", { method: "POST", body: JSON.stringify(payload) })} /></Overlay> : null}{modal.type === "new-customer" ? <Overlay><NewCustomerForm shopId={data.shop.id} saving={saving} onClose={() => setModal(null)} onSave={async (guardianPayload, petPayload) => { await mutate("/api/guardians", { method: "POST", body: JSON.stringify(guardianPayload) }); const refreshed = await fetchJson<BootstrapPayload>(`/api/bootstrap?shopId=${data.shop.id}`); setData(refreshed); const guardian = refreshed.guardians[refreshed.guardians.length - 1]; await mutate("/api/pets", { method: "POST", body: JSON.stringify({ ...petPayload, guardianId: guardian.id }) }); }} /></Overlay> : null}{modal.type === "add-pet" ? <Overlay><AddPetForm shopId={data.shop.id} guardianId={modal.guardianId} saving={saving} onClose={() => setModal(null)} onSave={(payload) => mutate("/api/pets", { method: "POST", body: JSON.stringify(payload) })} /></Overlay> : null}{modal.type === "edit-record" ? <Overlay><EditRecordForm shopId={data.shop.id} services={data.services} record={modal.record} saving={saving} onClose={() => setModal(null)} onSave={(payload) => mutate("/api/records", { method: "PATCH", body: JSON.stringify(payload) })} /></Overlay> : null}{modal.type === "stat" ? <Overlay><StatDetail kind={modal.kind} todayAppointments={todayConfirmedAppointments} pendingAppointments={pendingAppointments} overdueRows={revisitRows.filter((item) => item.status === "overdue")} estimatedRevenue={estimatedRevenue} petMap={petMap} guardianMap={guardianMap} serviceMap={serviceMap} saving={saving} onUpdate={(appointmentId: string, status: AppointmentStatus) => mutate("/api/appointments", { method: "PATCH", body: JSON.stringify({ appointmentId, status }) })} onClose={() => setModal(null)} /></Overlay> : null}</div>}
    </div>
  );
}

function Panel({ title, action, children }: { title: string; action?: string; children: React.ReactNode }) { return <section className="rounded-[28px] border border-[var(--border)] bg-white p-4 shadow-sm"><div className="mb-3 flex items-center justify-between"><h2 className="text-sm font-extrabold">{title}</h2>{action && <span className="text-xs text-[var(--muted)]">{action}</span>}</div><div className="space-y-3">{children}</div></section>; }
function StatCard({ label, value, tone, onClick }: { label: string; value: string; tone: "accent" | "warning" | "danger" | "neutral"; onClick: () => void }) { const toneMap = { accent: "border-[var(--accent)] bg-[var(--accent-soft)] text-[var(--accent)]", warning: "border-amber-200 bg-amber-50 text-amber-700", danger: "border-red-200 bg-red-50 text-red-700", neutral: "border-stone-200 bg-stone-50 text-stone-700" } as const; return <button onClick={onClick} className={`rounded-[26px] border p-4 text-left ${toneMap[tone]}`}><p className="text-[11px] font-bold">{label}</p><p className="mt-2 text-lg font-extrabold tracking-tight">{value}</p><p className="mt-2 text-[11px] opacity-80">{ownerHomeCopy.statTapHint}</p></button>; }
function AppointmentRow({ appointment, pet, guardian, service, onClick }: { appointment: Appointment; pet: Pet; guardian: BootstrapPayload["guardians"][number]; service: Service; onClick: () => void }) { return <button onClick={onClick} className="flex w-full items-center gap-3 rounded-2xl border border-[var(--border)] bg-[#fffdfa] px-3 py-3 text-left"><div className="min-w-12 text-sm font-extrabold">{appointment.appointment_time}</div><Avatar seed={pet.avatar_seed} /><div className="flex-1"><p className="text-sm font-bold">{pet.name} <span className="text-xs font-medium text-[var(--muted)]">({guardian.name})</span></p><p className="text-xs text-[var(--muted)]">{service.name} · {service.duration_minutes}분</p></div><span className="rounded-full px-2 py-1 text-[11px] font-bold" style={{ color: statusMeta[appointment.status].color, background: statusMeta[appointment.status].bg }}>{statusMeta[appointment.status].label}</span></button>; }
function AppointmentDetail({ appointment, pet, guardian, service, saving, onClose, onUpdate }: { appointment: Appointment; pet: Pet; guardian: BootstrapPayload["guardians"][number]; service: Service; saving: boolean; onClose: () => void; onUpdate: (status: AppointmentStatus) => void }) { const rollbackStatus = appointment.status === "almost_done" ? "confirmed" : appointment.status === "completed" ? "almost_done" : appointment.status === "cancelled" ? "confirmed" : null; const rollbackLabel = appointment.status === "almost_done" ? ownerHomeCopy.rollbackToConfirmed : appointment.status === "completed" ? ownerHomeCopy.rollbackToAlmostDone : appointment.status === "cancelled" ? ownerHomeCopy.restoreCancelled : null; return <Sheet title={ownerHomeCopy.appointmentDetailTitle} onClose={onClose}><div className="space-y-4"><div className="rounded-2xl bg-[#faf5f1] p-4 text-sm"><p className="font-bold">{pet.name} {ownerHomeCopy.separator} {guardian.name}</p><p className="mt-1 text-[var(--muted)]">{appointment.appointment_date} {appointment.appointment_time}</p><p className="mt-1 text-[var(--muted)]">{service.name} {ownerHomeCopy.separator} {won(service.price)}</p><p className="mt-1 text-[var(--muted)]">{ownerHomeCopy.memoLabel}: {appointment.memo || ownerHomeCopy.emptyMemo}</p></div><div className="grid grid-cols-2 gap-2">{appointment.status === "pending" && <><ActionButton onClick={() => onUpdate("confirmed")} disabled={saving}>{ownerHomeCopy.pendingApprove}</ActionButton><ActionButton onClick={() => onUpdate("rejected")} variant="secondary" disabled={saving}>{ownerHomeCopy.pendingReject}</ActionButton></>}{["confirmed", "in_progress"].includes(appointment.status) && <ActionButton variant="highlight" onClick={() => onUpdate("almost_done")} disabled={saving}>{ownerHomeCopy.fiveMinutesBefore}</ActionButton>}{appointment.status === "almost_done" && <ActionButton onClick={() => onUpdate("completed")} variant="secondary" disabled={saving}>{ownerHomeCopy.groomingComplete}</ActionButton>}{rollbackStatus && rollbackLabel && appointment.status !== "almost_done" && <ActionButton onClick={() => onUpdate(rollbackStatus)} variant="ghost" disabled={saving}>{rollbackLabel}</ActionButton>}</div></div></Sheet>; }

"bg-[var(--accent)] text-white" : variant === "secondary" ? "bg-[#f2ece7] text-[var(--text)]" : variant === "highlight" ? "border border-[#e3c0b3] bg-[#f9ede8] text-[var(--accent)] shadow-sm" : "border border-[var(--border)] bg-white text-[var(--muted)]"; return <button disabled={disabled} onClick={onClick} className={`w-full rounded-2xl px-4 py-3 text-sm font-bold disabled:opacity-50 ${className}`}>{children}</button>; }
function EmptyState({ title }: { title: string }) { return <div className="rounded-2xl border border-dashed border-[var(--border)] px-4 py-8 text-center text-sm text-[var(--muted)]">{title}</div>; }
function Avatar({ seed }: { seed: string }) { return <div className="flex size-11 items-center justify-center rounded-full bg-[var(--accent-soft)] text-lg">{seed}</div>; }
function UrgencyPill({ status, days }: { status: "overdue" | "soon" | "ok" | "unknown"; days: number | null }) { const text = status === "overdue" ? `${Math.abs(days || 0)}일 초과` : status === "soon" ? `${days}일 남음` : status === "ok" ? `${days}일 여유` : "미산정"; const cls = status === "overdue" ? "bg-red-50 text-red-700" : status === "soon" ? "bg-amber-50 text-amber-700" : "bg-emerald-50 text-emerald-700"; return <span className={`rounded-full px-2 py-1 text-[11px] font-bold ${cls}`}>{text}</span>; }
function InfoItem({ label, value }: { label: string; value: string }) { return <div className="rounded-2xl bg-[#faf5f1] p-3"><p className="text-[11px] font-bold text-[var(--muted)]">{label}</p><p className="mt-1 text-sm font-bold">{value}</p></div>; }
